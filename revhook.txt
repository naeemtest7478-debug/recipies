<?php
// Your webhook URL
$webhook_url = "https://webhook.site/f10f66fa-2954-4511-a30e-d6fbeecbe5c2/testrecipies";

// Data to send
$data = array(
    "message" => "Hello from PHP test script",
    "timestamp" => date("Y-m-d H:i:s")
);

// Convert data to JSON
$json_data = json_encode($data);

// Use file_get_contents with context for POST
$options = array(
    'http' => array(

    ),
);
